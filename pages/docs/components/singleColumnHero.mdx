```jsx showLineNumbers filename="sectioner/singleColumnHero/SingleColumnHero.jsx" copy
import { Link } from "react-router-dom";
import styles from "./styles.module.css";

const SingleColumnHero = ({
  heading,
  subHeading,
  buttonUrl,
  buttonText,
  className,
}) => {
  return (
    <div className={`${styles.hero} ${className}`}>
      <h1 className={styles.heading}>{heading}</h1>
      <p className={styles.subHeading}>{subHeading}</p>
      <Link className={styles.button} to={buttonUrl}>
        {buttonText}
      </Link>
    </div>
  );
};

export default SingleColumnHero;
```
